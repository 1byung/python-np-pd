#### 실습 코드
```python
from scipy.fft import fft, fftfreq
import numpy as np

# 오디오 로드
audio, sr = librosa.load('test.wav')

# FFT 수행
N = len(audio)
yf = fft(audio)
xf = fftfreq(N, 1/sr)

# 주파수 스펙트럼
plt.figure(figsize=(12, 4))
plt.plot(xf[:N//2], np.abs(yf[:N//2]))
plt.xlabel('Frequency (Hz)')
plt.ylabel('Magnitude')
plt.xlim(0, 5000)
plt.title('Frequency Spectrum')
plt.show()

# 지배적인 주파수 찾기 (노이즈 캔슬링 핵심!)
dominant_idx = np.argmax(np.abs(yf[:N//2]))
dominant_freq = xf[dominant_idx]
print(f"🎯 지배적인 주파수: {dominant_freq:.2f}Hz")
```

# 1. 간단한 예제: 사인파 노이즈 제거
sr = 22050
duration = 2.0
t = np.linspace(0, duration, int(sr * duration))

# 원본 신호 (깨끗한 음성)
clean = np.random.randn(len(t)) * 0.1

# 100Hz 노이즈 추가
noise = 0.5 * np.sin(2 * np.pi * 100 * t)
noisy = clean + noise

# 역위상 생성 (180도 반전!)
anti_noise = -noise

# 노이즈 제거
denoised = noisy + anti_noise
# = clean + noise + (-noise) = clean!

print(f"노이즈 있음: {np.std(noisy):.4f}")
print(f"노이즈 제거: {np.std(denoised):.4f}")


# 2. 실제 오디오에 적용
audio, sr = librosa.load('noisy_audio.wav')

# FFT로 주파수 찾기
yf = fft(audio)
xf = fftfreq(len(audio), 1/sr)

# 지배적 주파수 추출
dominant_idx = np.argmax(np.abs(yf[:len(yf)//2]))
dominant_freq = xf[dominant_idx]

# 역위상 생성
t = np.linspace(0, len(audio)/sr, len(audio))
anti_noise = -np.sin(2 * np.pi * dominant_freq * t)

# 진폭 조정 (중요!)
noise_amplitude = np.abs(yf[dominant_idx]) / len(yf)
anti_noise *= noise_amplitude

# 노이즈 제거
denoised = audio + anti_noise[:len(audio)]

# 저장
import soundfile as sf
sf.write('denoised_output.wav', denoised, sr)
print("✅ 노이즈 제거 완료!")
```

audio, sr = librosa.load('test.wav')
mfcc = librosa.feature.mfcc(y=audio, sr=sr, n_mfcc=13)

print(f"MFCC shape: {mfcc.shape}")  # (13, time_frames)

# MFCC 시각화
plt.figure(figsize=(12, 4))
librosa.display.specshow(mfcc, sr=sr, x_axis='time')
plt.colorbar()
plt.title('MFCC')
plt.show()

# 통계 (평균, 표준편차)
mfcc_mean = np.mean(mfcc, axis=1)
mfcc_std = np.std(mfcc, axis=1)


# 배치 처리 함수
def extract_features(file_path):
    """오디오 파일에서 특징 추출"""
    audio, sr = librosa.load(file_path, duration=5.0)
    mfcc = librosa.feature.mfcc(y=audio, sr=sr, n_mfcc=13)

    return {
        'mfcc_mean': np.mean(mfcc, axis=1),
        'mfcc_std': np.std(mfcc, axis=1),
        'spectral_centroid': np.mean(librosa.feature.spectral_centroid(y=audio, sr=sr))
    }

# 여러 파일 처리
import os
for filename in os.listdir('audio_data/'):
    if filename.endswith('.wav'):
        features = extract_features(f'audio_data/{filename}')
        print(f"{filename}: {features['mfcc_mean'][:3]}")
```

#### 미니 프로젝트: 간단한 노이즈 제거기
```python
# noise_remover.py
import librosa
import numpy as np
from scipy.fft import fft, fftfreq
import soundfile as sf

def remove_noise(input_file, output_file):
    """
    단일 주파수 노이즈 제거
    """
    # 1. 오디오 로드
    audio, sr = librosa.load(input_file)
    print(f"✅ 파일 로드: {len(audio)/sr:.2f}초")

    # 2. FFT로 주파수 분석
    yf = fft(audio)
    xf = fftfreq(len(audio), 1/sr)

    # 3. 지배적 주파수 찾기
    dominant_idx = np.argmax(np.abs(yf[:len(yf)//2]))
    dominant_freq = xf[dominant_idx]
    print(f"🎯 지배적 주파수: {dominant_freq:.2f}Hz")

    # 4. 역위상 생성
    t = np.linspace(0, len(audio)/sr, len(audio))
    anti_noise = -np.sin(2 * np.pi * dominant_freq * t)

    # 5. 진폭 조정
    noise_amp = np.abs(yf[dominant_idx]) / len(yf)
    anti_noise *= noise_amp

    # 6. 노이즈 제거
    denoised = audio + anti_noise[:len(audio)]

    # 7. 저장
    sf.write(output_file, denoised, sr)

    # 8. 결과 비교
    print(f"원본 RMS: {np.sqrt(np.mean(audio**2)):.4f}")
    print(f"제거 후 RMS: {np.sqrt(np.mean(denoised**2)):.4f}")
    print(f"✅ 저장 완료: {output_file}")

    return denoised, sr

# 테스트
if __name__ == "__main__":
    remove_noise('noisy_audio.wav', 'denoised_audio.wav')
```

## 💡 핵심 개념 요약

### 1. FFT (Fast Fourier Transform)
- 시간 도메인 → 주파수 도메인 변환

- 원래 파형의 180도 반전
- 소음 + 역위상 = 상쇄! (노이즈 캔슬링 핵심)
